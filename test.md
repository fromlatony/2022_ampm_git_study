### (h7)코드
~~~
#include <stdio.h>
//입력한 수까지의 세제곱 값을 출력하는 프로그램
int main() {
	int i, num;
	scanf_s("%d", &num);
	for (i = 1; i <= num; i++) {
		printf("%d\n", i * i * i);
	}
	return 0;
}
